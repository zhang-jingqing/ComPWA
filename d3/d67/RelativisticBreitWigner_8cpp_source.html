51 std::complex<double> RelativisticBreitWigner::dynamicalFunction(
52  double mSq, double mR, double ma, double mb, double width, unsigned int J,
53  double mesonRadius, formFactorType ffType) {
54 
55  std::complex<double> i(0, 1);
56  double sqrtS = sqrt(mSq);
57 
58  auto phspFactorSqrtS = phspFactor(sqrtS, ma, mb);
59  auto phspFactormR = phspFactor(mR, ma, mb);
60 
61  // Check if we have an event which is exactly at the phase space boundary
62  if (phspFactorSqrtS == std::complex<double>(0, 0))
63  return std::complex<double>(0, 0);
64 
65  std::complex<double> qTerm =
66  std::pow((phspFactorSqrtS / phspFactormR) * mR / sqrtS, (2 * J + 1));
67  double barrier = FormFactor(sqrtS, ma, mb, J, mesonRadius, ffType) /
68  FormFactor(mR, ma, mb, J, mesonRadius, ffType);
69 
70  // Calculate coupling constant to final state
71  std::complex<double> g_final =
72  widthToCoupling(mSq, mR, width, ma, mb, J, mesonRadius, ffType);
73 
74  // Coupling constant from production reaction. In case of a particle decay
75  // the production coupling doesn't depend in energy since the CM energy
76  // is in the (RC) system fixed to the mass of the decaying particle
77  double g_production = 1;
78 
79  std::complex<double> denom = std::complex<double>(mR * mR - mSq, 0) +
80  (-1.0) * i * sqrtS * (width * qTerm * barrier);
81 
82  std::complex<double> result = g_final * g_production / denom;
83 
84  assert(
85  (!std::isnan(result.real()) || !std::isinf(result.real())) &&
86  "RelativisticBreitWigner::dynamicalFunction() | Result is NaN or Inf!");
87  assert(
88  (!std::isnan(result.imag()) || !std::isinf(result.imag())) &&
89  "RelativisticBreitWigner::dynamicalFunction() | Result is NaN or Inf!");
90 
91  return result;
92 }

95 RelativisticBreitWigner::Factory(const boost::property_tree::ptree &pt) {
96  auto obj = std::make_shared<RelativisticBreitWigner>();
97 
98  std::string name = pt.get<std::string>("DecayParticle.<xmlattr>.Name");
99  LOG(trace) << "RelativisticBreitWigner::Factory() | Construction of " << name
100  << ".";
101  obj->SetName(name);
102  auto partProp = PhysConst::Instance()->FindParticle(name);
103  obj->SetMassParameter(
104  std::make_shared<DoubleParameter>(partProp.GetMassPar()));
105 
106  auto decayTr = partProp.GetDecayInfo();
107  if (partProp.GetDecayType() != "relativisticBreitWigner")
108  throw std::runtime_error(
109  "RelativisticBreitWigner::Factory() | Decay type does not match! ");
110 
111  auto spin = partProp.GetSpinQuantumNumber("Spin");
112  obj->SetSpin(spin);
113 
114  auto ffType = formFactorType(decayTr.get<int>("FormFactor.<xmlattr>.Type"));
115  obj->SetFormFactorType(ffType);
116 
117  // Read parameters from tree. Currently parameters of type 'Width' and
118  // 'MesonRadius' are required.
119  for (const auto &v : decayTr.get_child("")) {
120  if (v.first != "Parameter")
121  continue;
122  std::string type = v.second.get<std::string>("<xmlattr>.Type");
123  if (type == "Width") {
124  auto width = ComPWA::DoubleParameterFactory(v.second);
125  obj->SetWidthParameter(std::make_shared<DoubleParameter>(width));
126  } else if (type == "MesonRadius") {
127  auto mesonRadius = ComPWA::DoubleParameterFactory(v.second);
128  obj->SetMesonRadiusParameter(
129  std::make_shared<DoubleParameter>(mesonRadius));
130  } else {
131  throw std::runtime_error(
132  "RelativisticBreitWigner::Factory() | Parameter of type " + type +
133  " is unknown.");
134  }
135  }
136 
137  // Get masses of decay products
138  auto decayProducts = pt.get_child("DecayProducts");
139  if (decayProducts.size() != 2)
140  throw boost::property_tree::ptree_error(
141  "RelativisticBreitWigner::Factory() | Expect exactly two decay "
142  "products (" +
143  std::to_string(decayProducts.size()) + " given)!");
144 
145  auto firstItr = decayProducts.begin();
146  // auto secondItr = decayProducts.begin()+1; //compile error, no idea for why
147  auto secondItr = decayProducts.begin();
148  secondItr++;
149 
150  std::pair<std::string, std::string> daughterNames(
151  firstItr->second.get<std::string>("<xmlattr>.Name"),
152  secondItr->second.get<std::string>("<xmlattr>.Name"));
153  std::pair<double, double> daughterMasses(
154  PhysConst::Instance()->FindParticle(daughterNames.first).GetMass(),
155  PhysConst::Instance()->FindParticle(daughterNames.second).GetMass());
156  obj->SetDecayMasses(daughterMasses);
157  obj->SetDecayNames(daughterNames);
158 
159  LOG(trace)
160  << "RelativisticBreitWigner::Factory() | Construction of the decay "
161  << partProp.GetName() << " -> " << daughterNames.first << " + "
162  << daughterNames.second;
163 
164  return std::static_pointer_cast<AbstractDynamicalFunction>(obj);
165 }

236  if (paras.GetNMultiComplex() != check_nMComplex)
237  throw(BadParameter("BreitWignerStrat::execute() | "
238  "Number of MultiComplexes does not match: " +
239  std::to_string(paras.GetNMultiComplex()) +
240  " given but " + std::to_string(check_nMComplex) +
241  " expected."));
242 #endif
243 
248  double m0 = paras.GetDoubleParameter(0)->GetValue();
249  double Gamma0 = paras.GetDoubleParameter(1)->GetValue();
250  unsigned int spin = (unsigned int)paras.GetDoubleParameter(2)->GetValue();
251  double d = paras.GetDoubleParameter(3)->GetValue();
252  formFactorType ffType =
253  formFactorType(paras.GetDoubleParameter(4)->GetValue());
254  double ma = paras.GetDoubleParameter(5)->GetValue();
255  double mb = paras.GetDoubleParameter(6)->GetValue();
256 
257  std::vector<double> mp = paras.GetMultiDouble(0)->GetValues();
258 
259  std::vector<std::complex<double>> results(mp.size(),
260  std::complex<double>(0., 0.));
261 
262  // calc function for each point
263  for (unsigned int ele = 0; ele < mp.size(); ele++) {
264  try {
265  results.at(ele) = RelativisticBreitWigner::dynamicalFunction(
266  mp.at(ele), m0, ma, mb, Gamma0, spin, d, ffType);
267  } catch (std::exception &ex) {
268  LOG(error) << "BreitWignerStrategy::execute() | " << ex.what();
269  throw(std::runtime_error("BreitWignerStrategy::execute() | "
270  "Evaluation of dynamic function failed!"));
271  }
272  }
273  out =
274  std::shared_ptr<AbsParameter>(new MultiComplex(out->GetName(), results));
275  return true;
276 }

278 void RelativisticBreitWigner::GetParameters(ParameterList &list) {
279  AbstractDynamicalFunction::GetParameters(list);
280 
281  /* We check of for each parameter if a parameter of the same name exists in
282  * list. If so we check if both are equal and set the local parameter to the
283  * parameter from the list. In this way we connect parameters that occur on
284  * different positions in the amplitude.
285  */
286  std::shared_ptr<DoubleParameter> tmp, width, radius;
287  width = GetWidthParameter();
288  radius = GetMesonRadiusParameter();
289  try { // catch BadParameter
290  tmp = list.GetDoubleParameter(width->GetName());
291  try { // catch and throw std::runtime_error due to failed parameter
292  // comparisson
293  if (*tmp == *width)
294  SetWidthParameter(tmp);
295  } catch (std::exception &ex) {
296  throw;
297  }
298  } catch (BadParameter &ex) {
299  list.AddParameter(width);
300  }
301 
302  try { // catch BadParameter
303  tmp = list.GetDoubleParameter(radius->GetName());
304  try { // catch and throw std::runtime_error due to failed parameter
305  // comparisson
306  if (*tmp == *radius)
307  SetMesonRadiusParameter(tmp);
308  } catch (std::exception &ex) {
309  throw;
310  }
311  } catch (BadParameter &ex) {
312  list.AddParameter(radius);
313  }
314 }
