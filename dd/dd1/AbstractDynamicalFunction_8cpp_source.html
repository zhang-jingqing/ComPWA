12 void AbstractDynamicalFunction::GetParameters(ParameterList &list) {
13  // We check of for each parameter if a parameter of the same name exists in
14  // list. If so we check if both are equal and set the local parameter to the
15  // parameter from the list. In this way we connect parameters that occur on
16  // different positions in the amplitude.
17  std::shared_ptr<DoubleParameter> tmp;
18  auto mass = GetMassParameter();
19  try { // catch BadParameter
20  tmp = list.GetDoubleParameter(mass->GetName());
21  try { //catch and throw std::runtime_error due to failed parameter comparisson
22  if (*tmp == *mass)
23  SetMassParameter(tmp);
24  } catch (std::exception &ex) {
25  throw;
26  }
27  } catch (BadParameter &ex) {
28  list.AddParameter(mass);
29  }
30 }
