102 bool RunManager::gen(int number, std::shared_ptr<Kinematics> kin, std::shared_ptr<Generator> gen,
103  std::shared_ptr<AmpIntensity> amp,
104  std::shared_ptr<DataReader::Data> data,
105  std::shared_ptr<DataReader::Data> phsp,
106  std::shared_ptr<DataReader::Data> phspTrue) {
107 
108  if (number == 0)
109  return 0;
110 
111  // Doing some checks
112  if (number < 0 && !phsp)
113  throw std::runtime_error("RunManager: gen() negative number of events: " +
114  std::to_string((long double)number) +
115  ". And no phsp sample given!");
116  if (!amp)
117  throw std::runtime_error("RunManager::gen() | Amplitude not valid");
118  if (!gen)
119  throw std::runtime_error("RunManager::gen() | Generator not valid");
120  if (!data)
121  throw std::runtime_error("RunManager::gen() | Sample not valid");
122  if (data->GetNEvents() > 0)
123  throw std::runtime_error("RunManager::gen() | Sample not empty!");
124  if (phspTrue && !phsp)
125  throw std::runtime_error("RunManager::gen() | We have a sample of true"
126  " phsp events, but no phsp sample!");
127  if (phspTrue && phspTrue->GetNEvents() != phsp->GetNEvents())
128  throw std::runtime_error(
129  "RunManager::gen() | We have a sample of true "
130  "phsp events, but the sample size doesn't match that one of "
131  "the phsp sample!");
132 
133  double maxSampleWeight = 1.0;
134  if (phsp)
135  maxSampleWeight = phsp->GetMaxWeight();
136  if (phspTrue && phspTrue->GetMaxWeight() > maxSampleWeight)
137  maxSampleWeight = phspTrue->GetMaxWeight();
138 
139  /* Maximum value for random number generation. We introduce an arbitrary
140  * factor of 5 to make sure that the maximum value is never reached.
141  */
142  double generationMaxValue = 5* maxSampleWeight;
143 
144  unsigned int initialSeed = gen->GetSeed();
145  unsigned int totalCalls = 0;
146 
147  unsigned int limit;
148  if (phsp) {
149  limit = phsp->GetNEvents();
150  generationMaxValue *= Tools::Maximum(kin, amp, phsp) ;
151  } else {
152  limit = 100000000; // set large limit, should never be reached
153  }
154 
155  LOG(trace) << "RunMananger::gen() | Using " << generationMaxValue
156  << " as maximum value of the intensity.";
157 
158  Event evt; // event that we fill into generated sample
159  Event evtTrue; // event that is used to evalutate amplitude
160  progressBar bar(number);
161  if (number <= 0)
162  bar = progressBar(limit);
163  for (unsigned int i = 0; i < limit; i++) {
164  if (phsp && phspTrue) { // phsp and true sample is set
165  evtTrue = phspTrue->GetEvent(i);
166  evt = phsp->GetEvent(i);
167  } else if (phsp) { // phsp sample is set
168  evt = phsp->GetEvent(i);
169  evtTrue = evt;
170  } else { // otherwise generate event
171  gen->Generate(evt);
172  evtTrue = evt;
173  }
174  if (number <= 0)
175  bar.nextEvent();
176 
177  // use reconstructed position for weights
178  double weight = evt.GetWeight();
179 
180  // use true position for amplitude value
181  dataPoint point;
182  try {
183  kin->EventToDataPoint(evtTrue, point);
184  } catch (BeyondPhsp &ex) { // event outside phase, remove
185  continue;
186  }
187 
188  totalCalls++;
189  double ampRnd = gen->GetUniform(0, generationMaxValue);
190  double AMPpdf = amp->Intensity(point);
191 
192  // If maximum of intensity is reached we have to restart the procedure
193  if (generationMaxValue < (AMPpdf * weight)) {
194  LOG(trace) << "RunManager::gen() | Error in HitMiss "
195  "procedure: Maximum value of random number generation "
196  "smaller then amplitude maximum! We raise the maximum "
197  "value and restart generation!";
198  i = 0;
199  bar = progressBar(number);
200  gen->SetSeed(initialSeed);
201  generationMaxValue = 2 * (AMPpdf * weight);
202  data->Clear();
203  totalCalls = 0;
204  continue;
205  }
206 
207  if (ampRnd > (weight * AMPpdf))
208  continue;
209 
210  // Fill event to sample
211  /* reset weights: the weights are taken into account by hit and miss. The
212  * resulting
213  * sample is therefore unweighted */
214  evt.SetWeight(1.); // reset weight
215  evt.SetEfficiency(1.); // reset weight
216  data->PushEvent(evt);
217 
218  if (number > 0)
219  bar.nextEvent();
220  // break if we have a sufficienct number of events
221  if (data->GetNEvents() >= number)
222  i = limit;
223  }
224  if (data->GetNEvents() < number)
225  LOG(error) << "RunManager::gen() | Not able to generate "
226  << number << " events. Phsp sample too small. Current size "
227  "of sample is now " << data->GetNEvents();
228 
229  LOG(info) << "Efficiency of toy MC generation: "
230  << (double)data->GetNEvents() / totalCalls;
231 
232  return true;
233 }

235 bool RunManager::GeneratePhsp(int number) {
236  if (number == 0)
237  return 0;
238  if (!samplePhsp_)
239  throw std::runtime_error("RunManager: generatePhsp() | "
240  "No phase-space sample set");
241  if (samplePhsp_->GetNEvents() > 0)
242  throw std::runtime_error("RunManager: generatePhsp() | "
243  "Dataset not empty! abort!");
244 
245  LOG(info) << "Generating phase-space MC: [" << number << " events] ";
246 
247  progressBar bar(number);
248  for (unsigned int i = 0; i < number; i++) {
249  if (i > 0)
250  i--;
251  Event tmp;
252  gen_->Generate(tmp);
253  double ampRnd = gen_->GetUniform(0, 1);
254  if (ampRnd > tmp.GetWeight())
255  continue;
256 
257  /* Reset weights: weights are taken into account by hit&miss. The
258  * resulting sample is therefore unweighted
259  */
260  tmp.SetWeight(1.);
261 
262  tmp.SetEfficiency(1.);
263  i++;
264  samplePhsp_->PushEvent(tmp); // unfortunatly not thread safe
265  bar.nextEvent();
266  }
267  return true;
268 }
